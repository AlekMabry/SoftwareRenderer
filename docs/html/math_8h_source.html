233SGL_INLINE void sglVec2Barycentric(SGLVec2 p, SGLVec2 a, SGLVec2 b, SGLVec2 c,
234 float* u, float* v, float* w)
235{
236 SGLVec2 v0, v1, v2;
237 sglVec2Sub(b, a, v0);
238 sglVec2Sub(c, a, v1);
239 sglVec2Sub(p, a, v2);
240
241 float d00 = sglVec2Dot(v0, v0);
242 float d01 = sglVec2Dot(v0, v1);
243 float d11 = sglVec2Dot(v1, v1);
244 float d20 = sglVec2Dot(v2, v0);
245 float d21 = sglVec2Dot(v2, v1);
246 float denom = d00 * d11 - d01 * d01;
247 *v = (d11 * d20 - d01 * d21) / denom;
248 *w = (d00 * d21 - d01 * d20) / denom;
249 *u = 1.0f - *v - *w;
250}

345SGL_INLINE void sglMat4LookAt(SGLVec3 eye, SGLVec3 center, SGLVec3 up,
346 SGLMat4 dest)
347{
348 SGLVec3 f, u, s;
349 sglVec3Sub(center, eye, f);
350 sglVec3Normalize(f);
351
352 sglVec3Cross(f, up, s);
353 sglVec3Normalize(s);
354 sglVec3Cross(s, f, u);
355
356 dest[0][0] = s[0];
357 dest[0][1] = u[0];
358 dest[0][2] = -f[0];
359 dest[1][0] = s[1];
360 dest[1][1] = u[1];
361 dest[1][2] = -f[1];
362 dest[2][0] = s[2];
363 dest[2][1] = u[2];
364 dest[2][2] = -f[2];
365 dest[3][0] = -sglVec3Dot(s, eye);
366 dest[3][1] = -sglVec3Dot(u, eye);
367 dest[3][2] = sglVec3Dot(f, eye);
368 dest[0][3] = dest[1][3] = dest[2][3] = 0.0f;
369 dest[3][3] = 1.0f;
370}

375SGL_INLINE void sglMat4Perspective(float fovY, float aspect, float nearZ,
376 float farZ, SGLMat4 dest)
377{
378 // Right hand, [-1, 1] clipspace
379 float f, fn;
380
381 sglMat4Zero(dest);
382
383 f = 1.0f / tanf(fovY * 0.5f);
384 fn = 1.0f / (nearZ - farZ);
385
386 dest[0][0] = f / aspect;
387 dest[1][1] = f;
388 dest[2][2] = (nearZ + farZ) * fn;
389 dest[2][3] = -1.0f;
390 dest[3][2] = 2.0f * nearZ * farZ * fn;
391}
